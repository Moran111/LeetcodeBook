# 212. Word Search II

Given an `m x n` `board` of characters and a list of strings `words`, return _all words on the board_.

Each word must be constructed from letters of sequentially adjacent cells, where **adjacent cells** are horizontally or vertically neighboring. The same letter cell may not be used more than once in a word.

**Example 1:**![](https://assets.leetcode.com/uploads/2020/11/07/search1.jpg)

```
Input: board = [["o","a","a","n"],["e","t","a","e"],["i","h","k","r"],["i","f","l","v"]], words = ["oath","pea","eat","rain"]
Output: ["eat","oath"]
```

**Example 2:**![](https://assets.leetcode.com/uploads/2020/11/07/search2.jpg)

```
Input: board = [["a","b"],["c","d"]], words = ["abcb"]
Output: []
```

**Constraints:**

* `m == board.length`
* `n == board[i].length`
* `1 <= m, n <= 12`
* `board[i][j]` is a lowercase English letter.
* `1 <= words.length <= 3 * 104`
* `1 <= words[i].length <= 10`
* `words[i]` consists of lowercase English letters.
* All the strings of `words` are unique.



```
class Solution {
    
    Node root;
    Set<String> res = new HashSet<>();
    public List<String> findWords(char[][] board, String[] words) {
        root = new Node();
        for (String word: words) {
            insert(word);
        }
        
        boolean[][] visited = new boolean[board.length][board[0].length];
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                List<Character> list = new ArrayList<>();
                list.add(board[i][j]);
                visited[i][j] = true;
                // 进来dfs前，list里就有值了
                dfs(board, i, j, list, visited);
                visited[i][j] = false;
            }
        }
        return new ArrayList<>(res);
    }
    
    int[] dr = new int[]{1, 0, -1, 0};
    int[] dc = new int[]{0, 1, 0, -1};
    
    // backtracking
    private void dfs(char[][] board, int i, int j, List<Character> list, boolean[][] visited) {

        if (!search(list)) {
            return;
        } 
        if (hasWord(list)) {
            StringBuilder sb = new StringBuilder();
            for (char c: list) {
                sb.append(c);
            }
            res.add(sb.toString());
        }
        
        for (int d = 0; d < 4; ++d) {
            int newI = i + dr[d];
            int newJ = j + dc[d];
            if (newI < 0 || newJ < 0 || newI >= board.length || newJ >= board[0].length || visited[newI][newJ]) {
                continue;
            }
            list.add(board[newI][newJ]);
            visited[newI][newJ] = true;
            // 进来dfs前，list里就有值了
            dfs(board, newI, newJ, list, visited); 
            //只要一return 就会返回到dfs这里， 然后之后会删除掉
            list.remove(list.size()-1);
            visited[newI][newJ] = false;
        }       
    }
        
    // insert Trie
    private void insert(String word) {
        Node node = root;
        for (char c: word.toCharArray()) {
            Node[] child = node.children;
            // if c is not in trie
            if (child[c - 'a'] == null) {
                child[c -'a'] = new Node();
            }
            node = child[c -'a'];
        }
        node.isEnd = true;
    }
    
    // check if word(may be prefix or an entire word) exist in the Trie
    private boolean search(List<Character> word) {
        Node node = root;
        for (char c: word) {
            Node[] child = node.children;
            // if c is not in trie
            if (child[c - 'a'] == null) {
                return false;
            }
            node = child[c - 'a'];
        }
        return true;
    }
    
    // return true if it is a word but not prefix
    private boolean hasWord(List<Character> word) {
        Node node = root;
        for (char c: word) {
            Node[] child = node.children;
            // if c is not in trie
            if (child[c - 'a'] == null) {
                return false;
            }
            node = child[c - 'a'];
        }
        return node.isEnd;
    }
    
    class Node {
        Node[] children = new Node[26];
        boolean isEnd;
    }
}

```
Node root;
Set<String> res = new HashSet<>();
public List<String> findWords(char[][] board, String[] words) {
    root = new Node();
    for (String word: words) {
        insert(word);
    }

    boolean[][] visited = new boolean[board.length][board[0].length];
    for (int i = 0; i < board.length; i++) {
        for (int j = 0; j < board[0].length; j++) {
            dfs(board, i, j, new ArrayList<>(), visited);
        }
    }
    return new ArrayList<>(res);
}

// backtracking
private void dfs(char[][] board, int i, int j, ArrayList<Character> list, boolean[][] visited) {

    if (i < 0 || j < 0 || i >= board.length || j >= board[0].length) {
        return;
    }
    if (visited[i][j]) { // 这里不用backtrack list and set 因为list, set里没东西呢
        return;
    }
    visited[i][j] = true;
    list.add(board[i][j]);

    if (!search(list)) {
        // 这里需要backtrack 因为返回的时候会先返回44，然后没有backtrack就会继续
        // 执行dfs 45
        visited[i][j] = false;
        list.remove(list.size()-1);
        return;
    } 
    if (hasWord(list)) {
        StringBuilder sb = new StringBuilder();
        for (char c: list) {
            sb.append(c);
        }
        res.add(sb.toString());
    }

    dfs(board, i+1, j, list, visited); 
    dfs(board, i-1, j, list, visited); 
    dfs(board, i, j-1, list, visited);
    dfs(board, i, j+1, list, visited);
    visited[i][j] = false;
    list.remove(list.size()-1);
}

}
